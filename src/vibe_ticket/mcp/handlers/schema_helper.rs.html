1//! Helper functions for creating MCP tool schemas
2
3use serde_json::{Map, Value, json};
4use std::borrow::Cow;
5use std::sync::Arc;
6
7#[cfg(feature = "mcp")]
8use rmcp::model::Tool;
9
10#[cfg(not(feature = "mcp"))]
11use crate::mcp::handlers::common::Tool;
12
13/// Convert a JSON value to a Map for use as tool schema
14#[must_use]
15pub fn json_to_schema(value: Value) -> Map<String, Value> {
16    if let Value::Object(map) = value {
17        map
18    } else {
19        let mut map = Map::new();
20        map.insert("type".to_string(), json!("object"));
21        map.insert("properties".to_string(), Value::Object(Map::new()));
22        map
23    }
24}
25
26/// Create a basic MCP tool with the given parameters
27#[must_use]
28pub fn create_tool(name: &'static str, description: &'static str, schema: Value) -> Tool {
29    Tool {
30        name: Cow::Borrowed(name),
31        description: Some(Cow::Borrowed(description)),
32        input_schema: Arc::new(json_to_schema(schema)),
33        title: None,
34        output_schema: None,
35        icons: None,
36        annotations: None,
37        meta: None,
38    }
39}
