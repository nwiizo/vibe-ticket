13/// A time entry for a ticket
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub struct TimeEntry {
16    /// Unique entry ID
17    pub id: String,
18    /// Ticket ID
19    pub ticket_id: String,
20    /// Duration in minutes
21    pub duration_minutes: i64,
22    /// Notes about the work
23    pub notes: Option<String>,
24    /// Date of the work
25    pub date: DateTime<Utc>,
26    /// When the entry was created
27    pub created_at: DateTime<Utc>,
28}

30/// Active timer state
31#[derive(Debug, Clone, Serialize, Deserialize)]
32pub struct ActiveTimer {
33    /// Ticket ID being tracked
34    pub ticket_id: String,
35    /// Ticket slug for display
36    pub ticket_slug: String,
37    /// When the timer started
38    pub started_at: DateTime<Utc>,
39    /// Notes about the work
40    pub notes: Option<String>,
41}

43/// Time tracking data store
44#[derive(Debug, Default, Serialize, Deserialize)]
45pub struct TimeTracking {
46    /// Time entries by ticket ID
47    pub entries: HashMap<String, Vec<TimeEntry>>,
48    /// Currently active timer
49    pub active_timer: Option<ActiveTimer>,
50}
51
52impl TimeTracking {
53    /// Load time tracking data from file
54    pub fn load(project_dir: Option<&str>) -> Result<Self> {
55        let path = Self::data_path(project_dir)?;
56        if !path.exists() {
57            return Ok(Self::default());
58        }
59
60        let content = fs::read_to_string(&path).map_err(|e| {
61            VibeTicketError::custom(format!("Failed to read time tracking file: {e}"))
62        })?;
63        let data: Self = serde_yaml::from_str(&content).map_err(|e| {
64            VibeTicketError::custom(format!("Failed to parse time tracking file: {e}"))
65        })?;
66        Ok(data)
67    }
68
69    /// Save time tracking data to file
70    pub fn save(&self, project_dir: Option<&str>) -> Result<()> {
71        let path = Self::data_path(project_dir)?;
72        let content = serde_yaml::to_string(self).map_err(|e| {
73            VibeTicketError::custom(format!("Failed to serialize time tracking: {e}"))
74        })?;
75        fs::write(&path, content).map_err(|e| {
76            VibeTicketError::custom(format!("Failed to write time tracking file: {e}"))
77        })?;
78        Ok(())
79    }
80
81    /// Get the path to the time tracking file
82    fn data_path(project_dir: Option<&str>) -> Result<PathBuf> {
83        let project_root = find_project_root(project_dir)?;
84        Ok(project_root.join(".vibe-ticket").join("time_tracking.yaml"))
85    }
86
87    /// Add a time entry
88    pub fn add_entry(&mut self, entry: TimeEntry) {
89        self.entries
90            .entry(entry.ticket_id.clone())
91            .or_default()
92            .push(entry);
93    }
94
95    /// Get total time for a ticket in minutes
96    pub fn total_time_for_ticket(&self, ticket_id: &str) -> i64 {
97        self.entries
98            .get(ticket_id)
99            .map(|entries| entries.iter().map(|e| e.duration_minutes).sum())
100            .unwrap_or(0)
101    }
102}

104/// Parse time string like "1h30m", "2h", "45m" into minutes
105fn parse_time_string(time: &str) -> Result<i64> {
106    let time = time.to_lowercase();
107    let mut total_minutes: i64 = 0;
108    let mut current_num = String::new();
109
110    for c in time.chars() {
111        if c.is_ascii_digit() {
112            current_num.push(c);
113        } else if c == 'h' {
114            let hours: i64 = current_num
115                .parse()
116                .map_err(|_| VibeTicketError::custom(format!("Invalid time format: {time}")))?;
117            total_minutes += hours * 60;
118            current_num.clear();
119        } else if c == 'm' {
120            let minutes: i64 = current_num
121                .parse()
122                .map_err(|_| VibeTicketError::custom(format!("Invalid time format: {time}")))?;
123            total_minutes += minutes;
124            current_num.clear();
125        }
126    }
127
128    // If there's a remaining number without unit, treat as minutes
129    if !current_num.is_empty() {
130        let minutes: i64 = current_num
131            .parse()
132            .map_err(|_| VibeTicketError::custom(format!("Invalid time format: {time}")))?;
133        total_minutes += minutes;
134    }
135
136    if total_minutes == 0 {
137        return Err(VibeTicketError::custom(format!(
138            "Invalid time format: {time}. Use format like '1h30m', '2h', or '45m'"
139        )));
140    }
141
142    Ok(total_minutes)
143}

145/// Format minutes as human-readable string
146fn format_duration(minutes: i64) -> String {
147    let hours = minutes / 60;
148    let mins = minutes % 60;
149    if hours > 0 && mins > 0 {
150        format!("{hours}h {mins}m")
151    } else if hours > 0 {
152        format!("{hours}h")
153    } else {
154        format!("{mins}m")
155    }
156}

158/// Resolve ticket reference to ID and slug
159fn resolve_ticket(
160    ticket_ref: Option<String>,
161    project_dir: Option<&str>,
162) -> Result<(String, String)> {
163    let project_root = find_project_root(project_dir)?;
164    let vibe_ticket_dir = project_root.join(".vibe-ticket");
165    let storage = FileStorage::new(&vibe_ticket_dir);
166
167    let ticket_id = if let Some(ref_str) = ticket_ref {
168        crate::cli::handlers::common::resolve_ticket_ref(&storage, &ref_str)?
169    } else {
170        storage
171            .get_active()?
172            .ok_or(VibeTicketError::NoActiveTicket)?
173    };
174
175    let ticket = storage.load(&ticket_id)?;
176    Ok((ticket_id.to_string(), ticket.slug))
177}

179/// Handle time log command
180pub fn handle_time_log(
181    time: String,
182    ticket: Option<String>,
183    notes: Option<String>,
184    date: Option<String>,
185    project_dir: Option<&str>,
186    output: &OutputFormatter,
187) -> Result<()> {
188    let (ticket_id, ticket_slug) = resolve_ticket(ticket, project_dir)?;
189    let duration_minutes = parse_time_string(&time)?;
190
191    let entry_date = if let Some(date_str) = date {
192        // Parse date string - simplified for now
193        chrono::NaiveDate::parse_from_str(&date_str, "%Y-%m-%d")
194            .map_err(|_| {
195                VibeTicketError::custom(format!("Invalid date format: {date_str}. Use YYYY-MM-DD"))
196            })
197            .and_then(|d| {
198                d.and_hms_opt(12, 0, 0)
199                    .map(|dt| dt.and_utc())
200                    .ok_or_else(|| {
201                        VibeTicketError::custom("Failed to create date time".to_string())
202                    })
203            })?
204    } else {
205        Utc::now()
206    };
207
208    let entry = TimeEntry {
209        id: uuid::Uuid::new_v4().to_string(),
210        ticket_id: ticket_id.clone(),
211        duration_minutes,
212        notes: notes.clone(),
213        date: entry_date,
214        created_at: Utc::now(),
215    };
216
217    let mut tracking = TimeTracking::load(project_dir)?;
218    tracking.add_entry(entry);
219    tracking.save(project_dir)?;
220
221    let total = tracking.total_time_for_ticket(&ticket_id);
222
223    if output.is_json() {
224        output.print_json(&serde_json::json!({
225            "status": "success",
226            "ticket_id": ticket_id,
227            "ticket_slug": ticket_slug,
228            "logged": format_duration(duration_minutes),
229            "total": format_duration(total),
230        }))?;
231    } else {
232        output.success(&format!(
233            "Logged {} on ticket '{}'",
234            format_duration(duration_minutes),
235            ticket_slug
236        ));
237        if let Some(n) = notes {
238            output.info(&format!("Notes: {n}"));
239        }
240        output.info(&format!("Total time: {}", format_duration(total)));
241    }
242
243    Ok(())
244}

246/// Handle time start command
247pub fn handle_time_start(
248    ticket: Option<String>,
249    notes: Option<String>,
250    project_dir: Option<&str>,
251    output: &OutputFormatter,
252) -> Result<()> {
253    let (ticket_id, ticket_slug) = resolve_ticket(ticket, project_dir)?;
254
255    let mut tracking = TimeTracking::load(project_dir)?;
256
257    if tracking.active_timer.is_some() {
258        return Err(VibeTicketError::custom(
259            "Timer already running. Stop it first with 'vibe-ticket time stop'",
260        ));
261    }
262
263    tracking.active_timer = Some(ActiveTimer {
264        ticket_id,
265        ticket_slug: ticket_slug.clone(),
266        started_at: Utc::now(),
267        notes,
268    });
269
270    tracking.save(project_dir)?;
271
272    if output.is_json() {
273        output.print_json(&serde_json::json!({
274            "status": "success",
275            "action": "started",
276            "ticket_slug": ticket_slug,
277            "started_at": Utc::now().to_rfc3339(),
278        }))?;
279    } else {
280        output.success(&format!("Started timer for ticket '{ticket_slug}'"));
281        output.info(&format!("Started at: {}", Utc::now().format("%H:%M:%S")));
282    }
283
284    Ok(())
285}

287/// Handle time stop command
288pub fn handle_time_stop(
289    notes: Option<String>,
290    project_dir: Option<&str>,
291    output: &OutputFormatter,
292) -> Result<()> {
293    let mut tracking = TimeTracking::load(project_dir)?;
294
295    let timer = tracking
296        .active_timer
297        .take()
298        .ok_or_else(|| VibeTicketError::custom("No timer running"))?;
299
300    let duration = Utc::now().signed_duration_since(timer.started_at);
301    let duration_minutes = duration.num_minutes();
302
303    // Round up to at least 1 minute
304    let duration_minutes = if duration_minutes < 1 {
305        1
306    } else {
307        duration_minutes
308    };
309
310    let final_notes = notes.or(timer.notes);
311
312    let entry = TimeEntry {
313        id: uuid::Uuid::new_v4().to_string(),
314        ticket_id: timer.ticket_id.clone(),
315        duration_minutes,
316        notes: final_notes.clone(),
317        date: Utc::now(),
318        created_at: Utc::now(),
319    };
320
321    tracking.add_entry(entry);
322    tracking.save(project_dir)?;
323
324    let total = tracking.total_time_for_ticket(&timer.ticket_id);
325
326    if output.is_json() {
327        output.print_json(&serde_json::json!({
328            "status": "success",
329            "action": "stopped",
330            "ticket_slug": timer.ticket_slug,
331            "logged": format_duration(duration_minutes),
332            "total": format_duration(total),
333        }))?;
334    } else {
335        output.success(&format!("Stopped timer for ticket '{}'", timer.ticket_slug));
336        output.info(&format!(
337            "Time logged: {}",
338            format_duration(duration_minutes)
339        ));
340        if let Some(n) = final_notes {
341            output.info(&format!("Notes: {n}"));
342        }
343        output.info(&format!("Total time on ticket: {}", format_duration(total)));
344    }
345
346    Ok(())
347}

349/// Handle time status command
350pub fn handle_time_status(project_dir: Option<&str>, output: &OutputFormatter) -> Result<()> {
351    let tracking = TimeTracking::load(project_dir)?;
352
353    if let Some(timer) = &tracking.active_timer {
354        let elapsed = Utc::now().signed_duration_since(timer.started_at);
355        let elapsed_str = format_duration(elapsed.num_minutes().max(0));
356
357        if output.is_json() {
358            output.print_json(&serde_json::json!({
359                "status": "running",
360                "ticket_id": timer.ticket_id,
361                "ticket_slug": timer.ticket_slug,
362                "started_at": timer.started_at.to_rfc3339(),
363                "elapsed": elapsed_str,
364                "notes": timer.notes,
365            }))?;
366        } else {
367            output.success("Timer is running");
368            output.info(&format!("Ticket: {}", timer.ticket_slug));
369            output.info(&format!(
370                "Started: {}",
371                timer.started_at.format("%Y-%m-%d %H:%M:%S")
372            ));
373            output.info(&format!("Elapsed: {elapsed_str}"));
374            if let Some(notes) = &timer.notes {
375                output.info(&format!("Notes: {notes}"));
376            }
377        }
378    } else if output.is_json() {
379        output.print_json(&serde_json::json!({
380            "status": "stopped",
381        }))?;
382    } else {
383        output.info("No timer running");
384        output.info("");
385        output.info("Start one with:");
386        output.info("  vibe-ticket time start");
387    }
388
389    Ok(())
390}

392/// Handle time report command
393pub fn handle_time_report(
394    ticket: Option<String>,
395    all: bool,
396    _since: Option<String>,
397    _until: Option<String>,
398    project_dir: Option<&str>,
399    output: &OutputFormatter,
400) -> Result<()> {
401    let tracking = TimeTracking::load(project_dir)?;
402
403    if all {
404        // Show summary for all tickets
405        let mut total_minutes: i64 = 0;
406        let mut ticket_totals: Vec<(String, i64)> = Vec::new();
407
408        for (ticket_id, entries) in &tracking.entries {
409            let ticket_total: i64 = entries.iter().map(|e| e.duration_minutes).sum();
410            total_minutes += ticket_total;
411            ticket_totals.push((ticket_id.clone(), ticket_total));
412        }
413
414        ticket_totals.sort_by(|a, b| b.1.cmp(&a.1)); // Sort by time, descending
415
416        if output.is_json() {
417            let report: Vec<_> = ticket_totals
418                .iter()
419                .map(|(id, mins)| {
420                    serde_json::json!({
421                        "ticket_id": id,
422                        "total": format_duration(*mins),
423                        "minutes": mins,
424                    })
425                })
426                .collect();
427            output.print_json(&serde_json::json!({
428                "tickets": report,
429                "total": format_duration(total_minutes),
430                "total_minutes": total_minutes,
431            }))?;
432        } else {
433            output.info("Time Report (All Tickets)");
434            output.info(&format!("Total: {}", format_duration(total_minutes)));
435            output.info("");
436
437            for (ticket_id, mins) in ticket_totals {
438                // Get short ID
439                let short_id = if ticket_id.len() > 8 {
440                    &ticket_id[..8]
441                } else {
442                    &ticket_id
443                };
444                output.info(&format!("  {}: {}", short_id, format_duration(mins)));
445            }
446        }
447    } else {
448        // Show report for specific ticket
449        let (ticket_id, ticket_slug) = resolve_ticket(ticket, project_dir)?;
450
451        let entries = tracking
452            .entries
453            .get(&ticket_id)
454            .cloned()
455            .unwrap_or_default();
456        let total: i64 = entries.iter().map(|e| e.duration_minutes).sum();
457
458        if output.is_json() {
459            let entry_list: Vec<_> = entries
460                .iter()
461                .map(|e| {
462                    serde_json::json!({
463                        "id": e.id,
464                        "duration": format_duration(e.duration_minutes),
465                        "minutes": e.duration_minutes,
466                        "date": e.date.format("%Y-%m-%d").to_string(),
467                        "notes": e.notes,
468                    })
469                })
470                .collect();
471            output.print_json(&serde_json::json!({
472                "ticket_id": ticket_id,
473                "ticket_slug": ticket_slug,
474                "entries": entry_list,
475                "total": format_duration(total),
476                "total_minutes": total,
477            }))?;
478        } else {
479            output.info(&format!("Time Report for '{ticket_slug}'"));
480            output.info(&format!("Total: {}", format_duration(total)));
481            output.info("");
482
483            if entries.is_empty() {
484                output.info("No time entries");
485            } else {
486                for entry in entries.iter().rev().take(10) {
487                    let date = entry.date.format("%Y-%m-%d").to_string();
488                    let duration = format_duration(entry.duration_minutes);
489                    let notes = entry.notes.as_deref().unwrap_or("-");
490                    output.info(&format!("  {date} - {duration} - {notes}"));
491                }
492            }
493        }
494    }
495
496    Ok(())
497}
