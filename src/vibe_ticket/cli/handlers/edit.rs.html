12/// Parameters for updating a ticket
13struct UpdateParams<'a> {
14    title: Option<&'a str>,
15    description: Option<&'a str>,
16    priority: Option<&'a str>,
17    status: Option<&'a str>,
18    add_tags: Option<&'a str>,
19    remove_tags: Option<&'a str>,

22/// Handler for the `edit` command
23///
24/// This function allows editing various properties of a ticket:
25/// 1. Title
26/// 2. Description
27/// 3. Priority
28/// 4. Status
29/// 5. Tags (add/remove)
30/// 6. Opens in editor if requested
31///
32/// # Arguments
33///
34/// * `ticket_ref` - Optional ticket ID or slug (defaults to active ticket)
35/// * `title` - New title for the ticket
36/// * `description` - New description for the ticket
37/// * `priority` - New priority for the ticket
38/// * `status` - New status for the ticket
39/// * `add_tags` - Tags to add (comma-separated)
40/// * `remove_tags` - Tags to remove (comma-separated)
41/// * `editor` - Whether to open in the default editor
42/// * `project_dir` - Optional project directory path
43/// * `output` - Output formatter for displaying results
44///
45/// # Errors
46///
47/// Returns an error if:
48/// - The project is not initialized
49/// - No ticket is specified and there's no active ticket
50/// - The ticket is not found
51/// - Invalid priority or status values are provided
52#[allow(clippy::too_many_arguments, clippy::needless_pass_by_value)]
53pub fn handle_edit_command(
54    ticket_ref: Option<String>,
55    title: Option<String>,
56    description: Option<String>,
57    priority: Option<String>,
58    status: Option<String>,
59    add_tags: Option<String>,
60    remove_tags: Option<String>,
61    editor: bool,
62    project_dir: Option<&str>,
63    output: &OutputFormatter,
64) -> Result<()> {
65    // Ensure project is initialized
66    let project_root = find_project_root(project_dir)?;
67    let vibe_ticket_dir = project_root.join(".vibe-ticket");
68
69    // Initialize storage
70    let storage = FileStorage::new(&vibe_ticket_dir);
71
72    // Get the active ticket if no ticket specified
73    let ticket_id = if let Some(ref_str) = ticket_ref {
74        resolve_ticket_ref(&storage, &ref_str)?
75    } else {
76        // Get active ticket
77        storage
78            .get_active()?
79            .ok_or(VibeTicketError::NoActiveTicket)?
80    };
81
82    // Load the ticket
83    let mut ticket = storage.load(&ticket_id)?;
84
85    // Track what was changed
86    let mut changes = Vec::new();
87
88    // Open in editor if requested
89    if editor {
90        edit_in_editor(&mut ticket, &storage, output)?;
91        return Ok(());
92    }
93
94    // Apply updates
95    let update_params = UpdateParams {
96        title: title.as_deref(),
97        description: description.as_deref(),
98        priority: priority.as_deref(),
99        status: status.as_deref(),
100        add_tags: add_tags.as_deref(),
101        remove_tags: remove_tags.as_deref(),
102    };
103    apply_ticket_updates(&mut ticket, &mut changes, &update_params)?;
104
105    // Check if any changes were made
106    if changes.is_empty() {
107        output.warning("No changes specified");
108        return Ok(());
109    }
110
111    // Save the updated ticket
112    storage.save(&ticket)?;
113
114    // Notify MCP about ticket update
115    #[cfg(feature = "mcp")]
116    crate::integration::notify_ticket_updated(&ticket);
117
118    // Output results
119    format_edit_output(output, &ticket, &changes)?;
120
121    Ok(())
122}

124/// Apply updates to a ticket
125fn apply_ticket_updates(
126    ticket: &mut crate::core::Ticket,
127    changes: &mut Vec<String>,
128    params: &UpdateParams<'_>,
129) -> Result<()> {
130    // Update title if provided
131    if let Some(new_title) = params.title {
132        let old_title = ticket.title.clone();
133        ticket.title = new_title.to_string();
134        changes.push(format!("Title: {old_title} → {new_title}"));
135    }
136
137    // Update description if provided
138    if let Some(new_description) = params.description {
139        ticket.description = new_description.to_string();
140        changes.push("Description updated".to_string());
141    }
142
143    // Update priority if provided
144    if let Some(priority_str) = params.priority {
145        let new_priority =
146            Priority::try_from(priority_str).map_err(|_| VibeTicketError::InvalidPriority {
147                priority: priority_str.to_string(),
148            })?;
149        let old_priority = ticket.priority;
150        ticket.priority = new_priority;
151        changes.push(format!("Priority: {old_priority} → {new_priority}"));
152    }
153
154    // Update status if provided
155    if let Some(status_str) = params.status {
156        let new_status =
157            Status::try_from(status_str).map_err(|_| VibeTicketError::InvalidStatus {
158                status: status_str.to_string(),
159            })?;
160        let old_status = ticket.status;
161        ticket.status = new_status;
162        changes.push(format!("Status: {old_status} → {new_status}"));
163
164        // Update timestamps based on status changes
165        match (old_status, new_status) {
166            (Status::Todo, Status::Doing) => {
167                ticket.started_at = Some(chrono::Utc::now());
168            },
169            (_, Status::Done) if old_status != Status::Done => {
170                ticket.closed_at = Some(chrono::Utc::now());
171            },
172            _ => {},
173        }
174    }
175
176    // Handle tags
177    handle_tag_updates(ticket, changes, params.add_tags, params.remove_tags);
178
179    Ok(())
180}

182/// Handle tag updates
183fn handle_tag_updates(
184    ticket: &mut crate::core::Ticket,
185    changes: &mut Vec<String>,
186    add_tags: Option<&str>,
187    remove_tags: Option<&str>,
188) {
189    // Add tags if provided
190    if let Some(tags_str) = add_tags {
191        let new_tags: Vec<String> = tags_str
192            .split(',')
193            .map(|s| s.trim().to_string())
194            .filter(|s| !s.is_empty())
195            .collect();
196
197        for tag in new_tags {
198            if !ticket.tags.contains(&tag) {
199                ticket.tags.push(tag);
200            }
201        }
202        changes.push("Tags added".to_string());
203    }
204
205    // Remove tags if provided
206    if let Some(tags_str) = remove_tags {
207        let tags_to_remove: Vec<String> = tags_str
208            .split(',')
209            .map(|s| s.trim().to_string())
210            .filter(|s| !s.is_empty())
211            .collect();
212
213        ticket.tags.retain(|tag| !tags_to_remove.contains(tag));
214        changes.push("Tags removed".to_string());
215    }
216}

218/// Format and display edit command output
219fn format_edit_output(
220    output: &OutputFormatter,
221    ticket: &crate::core::Ticket,
222    changes: &[String],
223) -> Result<()> {
224    if output.is_json() {
225        output.print_json(&serde_json::json!({
226            "status": "success",
227            "ticket": {
228                "id": ticket.id.to_string(),
229                "slug": ticket.slug,
230                "title": ticket.title,
231                "description": ticket.description,
232                "status": ticket.status.to_string(),
233                "priority": ticket.priority.to_string(),
234                "tags": ticket.tags,
235            },
236            "changes": changes,
237        }))?;
238    } else {
239        output.success(&format!("Updated ticket: {}", ticket.slug));
240        for change in changes {
241            output.info(&format!("  • {change}"));
242        }
243
244        // Show current state
245        output.info("");
246        output.info("Current state:");
247        output.info(&format!("  Title: {}", ticket.title));
248        output.info(&format!("  Status: {}", ticket.status));
249        output.info(&format!("  Priority: {}", ticket.priority));
250        if !ticket.tags.is_empty() {
251            output.info(&format!("  Tags: {}", ticket.tags.join(", ")));
252        }
253    }
254    Ok(())
255}

257/// Edit ticket in the default editor
258fn edit_in_editor(
259    ticket: &mut crate::core::Ticket,
260    storage: &FileStorage,
261    output: &OutputFormatter,
262) -> Result<()> {
263    use std::io::Write as IoWrite;
264    use std::process::Command;
265
266    // Create a temporary file with the ticket content
267    let temp_dir = std::env::temp_dir();
268    let temp_file = temp_dir.join(format!("vibe-ticket-{}.yaml", ticket.id));
269
270    // Serialize ticket to YAML
271    let yaml_content = serde_yaml::to_string(&ticket)
272        .map_err(|e| VibeTicketError::custom(format!("Failed to serialize YAML: {e}")))?;
273
274    // Write to temporary file
275    let mut file = std::fs::File::create(&temp_file)
276        .map_err(|e| VibeTicketError::io_error("create", &temp_file, e))?;
277    file.write_all(yaml_content.as_bytes())
278        .map_err(|e| VibeTicketError::io_error("write", &temp_file, e))?;
279
280    // Get editor from environment
281    let editor = std::env::var("EDITOR").unwrap_or_else(|_| "vi".to_string());
282
283    // Open editor
284    let status = Command::new(&editor)
285        .arg(&temp_file)
286        .status()
287        .map_err(|e| VibeTicketError::custom(format!("Failed to launch editor: {e}")))?;
288
289    if !status.success() {
290        return Err(VibeTicketError::custom("Editor exited with error"));
291    }
292
293    // Read the edited content
294    let edited_content = std::fs::read_to_string(&temp_file)
295        .map_err(|e| VibeTicketError::io_error("read", &temp_file, e))?;
296
297    // Parse the edited ticket
298    let edited_ticket: crate::core::Ticket = serde_yaml::from_str(&edited_content)
299        .map_err(|e| VibeTicketError::deserialization_error("YAML ticket", e))?;
300
301    // Update the original ticket
302    *ticket = edited_ticket;
303
304    // Save the updated ticket
305    storage.save(ticket)?;
306
307    // Clean up temp file
308    let _ = std::fs::remove_file(&temp_file);
309
310    output.success(&format!("Updated ticket: {}", ticket.slug));
311
312    Ok(())
313}
