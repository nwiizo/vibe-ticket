12/// Handler for the `check` command
13///
14/// This function displays:
15/// 1. Project information
16/// 2. Active ticket details (if any)
17/// 3. Current Git branch
18/// 4. Project statistics (optional)
19/// 5. Recent tickets (in detailed mode)
20///
21/// # Arguments
22///
23/// * `detailed` - Whether to show detailed information
24/// * `stats` - Whether to include statistics
25/// * `project_dir` - Optional project directory path
26/// * `output` - Output formatter for displaying results
27///
28/// # Errors
29///
30/// Returns an error if:
31/// - The project is not initialized
32/// - File I/O operations fail
33pub fn handle_check_command(
34    detailed: bool,
35    stats: bool,
36    project_dir: Option<&str>,
37    output: &OutputFormatter,
38) -> Result<()> {
39    // Gather all data
40    let check_data = gather_check_data(detailed, stats, project_dir)?;
41
42    // Output results
43    if output.is_json() {
44        output_json(&check_data, output)?;
45    } else {
46        output_text(&check_data, detailed, output);
47    }
48
49    Ok(())
50}

52/// Data structure for check command
53struct CheckData {
54    project_root: std::path::PathBuf,
55    project_state: crate::storage::ProjectState,
56    active_ticket: Option<Ticket>,
57    current_branch: Option<String>,
58    statistics: Option<Statistics>,
59    recent_tickets: Vec<Ticket>,
60}

62/// Gather all data needed for check command
63fn gather_check_data(detailed: bool, stats: bool, project_dir: Option<&str>) -> Result<CheckData> {
64    let project_root = find_project_root(project_dir)?;
65    let vibe_ticket_dir = project_root.join(".vibe-ticket");
66    let storage = FileStorage::new(&vibe_ticket_dir);
67
68    let project_state = storage.load_state()?;
69    let active_ticket_id = storage.get_active()?;
70    let active_ticket = if let Some(id) = &active_ticket_id {
71        Some(storage.load(id)?)
72    } else {
73        None
74    };
75
76    let current_branch = get_current_git_branch(&project_root);
77    let statistics = if stats || detailed {
78        Some(calculate_statistics(&storage)?)
79    } else {
80        None
81    };
82
83    let recent_tickets = if detailed {
84        get_recent_tickets(&storage, 5)?
85    } else {
86        vec![]
87    };
88
89    Ok(CheckData {
90        project_root,
91        project_state,
92        active_ticket,
93        current_branch,
94        statistics,
95        recent_tickets,
96    })
97}

99/// Output check data as JSON
100fn output_json(data: &CheckData, output: &OutputFormatter) -> Result<()> {
101    output.print_json(&serde_json::json!({
102        "project": {
103            "name": data.project_state.name,
104            "description": data.project_state.description,
105            "created_at": data.project_state.created_at,
106            "path": data.project_root,
107        },
108        "active_ticket": data.active_ticket.as_ref().map(|t| serde_json::json!({
109            "id": t.id.to_string(),
110            "slug": t.slug,
111            "title": t.title,
112            "status": t.status.to_string(),
113            "priority": t.priority.to_string(),
114            "started_at": t.started_at,
115        })),
116        "git_branch": data.current_branch,
117        "statistics": data.statistics,
118        "recent_tickets": data.recent_tickets.iter().map(|t| serde_json::json!({
119            "id": t.id.to_string(),
120            "slug": t.slug,
121            "title": t.title,
122            "status": t.status.to_string(),
123        })).collect::<Vec<_>>(),
124    }))
125}

127/// Output check data as text
128fn output_text(data: &CheckData, detailed: bool, output: &OutputFormatter) {
129    // Project info
130    output.info(&format!("Project: {}", data.project_state.name));
131    if let Some(desc) = &data.project_state.description {
132        output.info(&format!("Description: {desc}"));
133    }
134    output.info(&format!("Path: {}", data.project_root.display()));
135    output.info(&format!(
136        "Created: {}",
137        format_datetime(data.project_state.created_at)
138    ));
139
140    if let Some(branch) = &data.current_branch {
141        output.info(&format!("Git branch: {branch}"));
142    }
143
144    output.info("");
145
146    // Active ticket
147    if let Some(ticket) = &data.active_ticket {
148        display_active_ticket(ticket, output);
149    } else {
150        output.info("No active ticket");
151    }
152
153    // Statistics
154    if let Some(stats) = &data.statistics {
155        display_statistics(stats, detailed, output);
156    }
157
158    // Recent tickets
159    if detailed && !data.recent_tickets.is_empty() {
160        display_recent_tickets(&data.recent_tickets, output);
161    }
162}

164/// Display active ticket information
165fn display_active_ticket(ticket: &Ticket, output: &OutputFormatter) {
166    output.success("Active Ticket:");
167    output.info(&format!("  ID: {}", ticket.id));
168    output.info(&format!("  Slug: {}", ticket.slug));
169    output.info(&format!("  Title: {}", ticket.title));
170    output.info(&format!("  Status: {}", ticket.status));
171    output.info(&format!("  Priority: {}", ticket.priority));
172
173    if let Some(started_at) = ticket.started_at {
174        let duration = Utc::now() - started_at;
175        let hours = duration.num_hours();
176        let minutes = duration.num_minutes() % 60;
177        output.info(&format!("  Time spent: {hours}h {minutes}m"));
178    }
179
180    if !ticket.tasks.is_empty() {
181        let completed = ticket.tasks.iter().filter(|t| t.completed).count();
182        output.info(&format!("  Tasks: {}/{}", completed, ticket.tasks.len()));
183    }
184}

226/// Project statistics
227#[derive(Debug, serde::Serialize)]
228struct Statistics {
229    total: usize,
230    todo: usize,
231    doing: usize,
232    review: usize,
233    blocked: usize,
234    done: usize,
235    critical: usize,
236    high: usize,
237    medium: usize,
238    low: usize,
239}

241/// Calculate project statistics
242fn calculate_statistics(storage: &FileStorage) -> Result<Statistics> {
243    let tickets = storage.load_all()?;
244
245    let mut stats = Statistics {
246        total: tickets.len(),
247        todo: 0,
248        doing: 0,
249        review: 0,
250        blocked: 0,
251        done: 0,
252        critical: 0,
253        high: 0,
254        medium: 0,
255        low: 0,
256    };
257
258    for ticket in &tickets {
259        // Count by status
260        match ticket.status {
261            Status::Todo => stats.todo += 1,
262            Status::Doing => stats.doing += 1,
263            Status::Review => stats.review += 1,
264            Status::Blocked => stats.blocked += 1,
265            Status::Done => stats.done += 1,
266        }
267
268        // Count by priority
269        match ticket.priority {
270            crate::core::Priority::Critical => stats.critical += 1,
271            crate::core::Priority::High => stats.high += 1,
272            crate::core::Priority::Medium => stats.medium += 1,
273            crate::core::Priority::Low => stats.low += 1,
274        }
275    }
276
277    Ok(stats)
278}

280/// Get recent tickets sorted by creation date
281fn get_recent_tickets(storage: &FileStorage, limit: usize) -> Result<Vec<Ticket>> {
282    let mut tickets = storage.load_all()?;
283
284    // Sort by creation date (descending)
285    tickets.sort_by(|a, b| b.created_at.cmp(&a.created_at));
286
287    // Take the specified limit
288    tickets.truncate(limit);
289
290    Ok(tickets)
291}

293/// Get current Git branch name
294fn get_current_git_branch(project_root: &std::path::Path) -> Option<String> {
295    use std::process::Command;
296
297    let output = Command::new("git")
298        .arg("rev-parse")
299        .arg("--abbrev-ref")
300        .arg("HEAD")
301        .current_dir(project_root)
302        .output()
303        .ok()?;
304
305    if output.status.success() {
306        Some(String::from_utf8_lossy(&output.stdout).trim().to_string())
307    } else {
308        None
309    }
310}

312/// Format datetime for display
313fn format_datetime(dt: DateTime<Utc>) -> String {
314    dt.with_timezone(&Local)
315        .format("%Y-%m-%d %H:%M")
316        .to_string()
317}
