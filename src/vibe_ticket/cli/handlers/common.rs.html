8/// Common context for all handler operations
9pub struct HandlerContext {
10    pub storage: FileStorage,
11}
12
13impl HandlerContext {
14    /// Create a new handler context
15    pub fn new(project_dir: Option<&str>) -> Result<Self> {
16        let project_root = find_project_root(project_dir)?;
17        let vibe_ticket_dir = project_root.join(".vibe-ticket");
18        let storage = FileStorage::new(&vibe_ticket_dir);

20        Ok(Self { storage })
21    }
22
23    /// Get storage reference
24    #[allow(dead_code)]
25    pub const fn storage(&self) -> &FileStorage {

27    }

30/// Common trait for ticket operations
31pub trait TicketOperation {
32    /// Load a ticket by reference (ID, slug, or active)
33    fn load_ticket(&self, ticket_ref: Option<&str>) -> Result<Ticket>;
34
35    /// Save a ticket
36    fn save_ticket(&self, ticket: &Ticket) -> Result<()>;
37
38    /// Resolve ticket reference to ID
39    fn resolve_ticket_ref(&self, ticket_ref: &str) -> Result<TicketId>;
40
41    /// Get active ticket ID
42    fn get_active_ticket_id(&self) -> Result<TicketId>;
43}
44
45impl TicketOperation for HandlerContext {
46    fn load_ticket(&self, ticket_ref: Option<&str>) -> Result<Ticket> {
47        let ticket_id = if let Some(ref_str) = ticket_ref {
48            self.resolve_ticket_ref(ref_str)?
49        } else {
50            self.get_active_ticket_id()?
51        };
52
53        self.storage.load_ticket(&ticket_id)
54    }
55
56    fn save_ticket(&self, ticket: &Ticket) -> Result<()> {
57        self.storage.save_ticket(ticket)
58    }
59
60    fn resolve_ticket_ref(&self, ticket_ref: &str) -> Result<TicketId> {
61        // Try to parse as UUID first
62        if let Ok(id) = Uuid::parse_str(ticket_ref) {
63            return Ok(TicketId::from_uuid(id));
64        }
65
66        // Try to find by slug
67        let tickets = self.storage.load_all_tickets()?;
68        for ticket in tickets {
69            if ticket.slug == ticket_ref {
70                return Ok(ticket.id);
71            }
72        }
73
74        Err(VibeTicketError::TicketNotFound {
75            id: ticket_ref.to_string(),
76        })
77    }
78
79    fn get_active_ticket_id(&self) -> Result<TicketId> {
80        self.storage
81            .get_active()?
82            .ok_or(VibeTicketError::NoActiveTicket)
83    }
84}
