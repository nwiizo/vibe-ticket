14/// Context for handler operations
15///
16/// Encapsulates common initialization and resources needed by handlers
17pub struct HandlerContext {
18    pub project_root: PathBuf,
19    pub tickets_dir: PathBuf,
20    pub storage: FileStorage,
21    pub formatter: OutputFormatter,
22}
23
24impl HandlerContext {
25    /// Create a new handler context
26    ///
27    /// Handles project directory resolution, storage initialization,
28    /// and common error checking.
29    ///
30    /// # Errors
31    ///
32    /// Returns an error if:
33    /// - Failed to change to the project directory
34    /// - Failed to get the current directory
35    /// - Project is not initialized (.vibe-ticket directory doesn't exist)
36    pub fn new(project_dir: Option<&str>, formatter: OutputFormatter) -> Result<Self> {
37        // Change to project directory if specified
38        if let Some(project_path) = project_dir {
39            env::set_current_dir(project_path)?;
40        }
41
42        let current_dir = env::current_dir()?;
43        let project_root = utils::find_project_root(current_dir.to_str())?;
44        let tickets_dir = project_root.join(".vibe-ticket");
45
46        if !tickets_dir.exists() {
47            return Err(VibeTicketError::ProjectNotInitialized);
48        }
49
50        let storage = FileStorage::new(tickets_dir.clone());
51
52        Ok(Self {
53            project_root,
54            tickets_dir,
55            storage,
56            formatter,
57        })
58    }
59
60    /// Get the currently active ticket ID
61    ///
62    /// # Errors
63    ///
64    /// Returns an error if:
65    /// - Failed to read the active ticket file
66    /// - No active ticket is set
67    pub fn get_active_ticket_id(&self) -> Result<TicketId> {
68        let active_ticket_path = self.tickets_dir.join("active_ticket");
69
70        if !active_ticket_path.exists() {
71            return Err(VibeTicketError::Custom(
72                "No active ticket. Use 'vibe-ticket work-on' to select a ticket.".to_string(),
73            ));
74        }
75
76        let content = std::fs::read_to_string(&active_ticket_path)?;
77        let id_str = content.trim();
78
79        TicketId::parse_str(id_str)
80            .map_err(|_| VibeTicketError::Custom(format!("Invalid active ticket ID: {id_str}")))
81    }
82
83    /// Set the active ticket
84    ///
85    /// # Errors
86    ///
87    /// Returns an error if failed to write the active ticket file
88    pub fn set_active_ticket(&self, ticket_id: &TicketId) -> Result<()> {
89        let active_ticket_path = self.tickets_dir.join("active_ticket");
90        std::fs::write(&active_ticket_path, ticket_id.to_string())?;
91        Ok(())
92    }
93
94    /// Clear the active ticket
95    ///
96    /// # Errors
97    ///
98    /// Returns an error if failed to remove the active ticket file
99    pub fn clear_active_ticket(&self) -> Result<()> {
100        let active_ticket_path = self.tickets_dir.join("active_ticket");
101        if active_ticket_path.exists() {
102            std::fs::remove_file(&active_ticket_path)?;
103        }
104        Ok(())
105    }
106
107    /// Resolve a ticket reference (ID, slug, or active)
108    ///
109    /// # Errors
110    ///
111    /// Returns an error if:
112    /// - The ticket reference is invalid
113    /// - The ticket is not found by ID or slug
114    /// - No active ticket is set when reference is None
115    pub fn resolve_ticket_ref(&self, ticket_ref: Option<&str>) -> Result<TicketId> {
116        if let Some(ref_str) = ticket_ref {
117            // Try to parse as ID first
118            if let Ok(id) = TicketId::parse_str(ref_str) {
119                return Ok(id);
120            }
121
122            // Try to find by slug
123            let tickets = self.storage.load_all()?;
124            for ticket in tickets {
125                if ticket.slug == ref_str {
126                    return Ok(ticket.id);
127                }
128            }
129
130            Err(VibeTicketError::Custom(format!(
131                "Ticket not found: {ref_str}"
132            )))
133        } else {
134            // Get active ticket
135            self.get_active_ticket_id()
136        }
137    }
138
139    /// Load a ticket by reference
140    ///
141    /// # Errors
142    ///
143    /// Returns an error if:
144    /// - Failed to resolve the ticket reference
145    /// - Failed to load the ticket from storage
146    pub fn load_ticket_by_ref(&self, ticket_ref: Option<&str>) -> Result<Ticket> {
147        let ticket_id = self.resolve_ticket_ref(ticket_ref)?;
148        self.storage.load(&ticket_id)
149    }

172/// Common validation functions
173pub mod validation {
174    use crate::core::Priority;
175    use crate::error::{Result, VibeTicketError};
176
177    /// Validate and parse priority string
178    ///
179    /// # Errors
180    ///
181    /// Returns an error if the priority string is not one of: low, medium, high, critical
182    pub fn parse_priority(priority: &str) -> Result<Priority> {
183        match priority.to_lowercase().as_str() {
184            "low" => Ok(Priority::Low),
185            "medium" => Ok(Priority::Medium),
186            "high" => Ok(Priority::High),
187            "critical" => Ok(Priority::Critical),
188            _ => Err(VibeTicketError::Custom(format!(
189                "Invalid priority: {priority}. Must be one of: low, medium, high, critical"
190            ))),
191        }
192    }
193
194    /// Validate ticket title
195    ///
196    /// # Errors
197    ///
198    /// Returns an error if:
199    /// - The title is empty
200    /// - The title exceeds 200 characters
201    pub fn validate_title(title: &str) -> Result<()> {
202        if title.trim().is_empty() {
203            return Err(VibeTicketError::Custom(
204                "Ticket title cannot be empty".to_string(),
205            ));
206        }
207        if title.len() > 200 {
208            return Err(VibeTicketError::Custom(
209                "Ticket title cannot exceed 200 characters".to_string(),
210            ));
211        }
212        Ok(())
213    }
214
215    /// Validate tags
216    #[must_use]
217    pub fn parse_tags(tags_str: Option<&str>) -> Vec<String> {
218        tags_str
219            .map(|s| {
220                s.split(',')
221                    .map(|tag| tag.trim().to_string())
222                    .filter(|tag| !tag.is_empty())
223                    .collect()
224            })
225            .unwrap_or_default()
226    }
227}
